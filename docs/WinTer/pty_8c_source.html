23 saAttr.bInheritHandle = TRUE;
24 saAttr.lpSecurityDescriptor = NULL;
25
26 if (!CreatePipe(&hInputRead, &hInputWrite, &saAttr, 0)) {
27 fprintf(stderr, "[PTY] Failed to create input pipe. Error: %lu\n", GetLastError());

29 }
30
31 if (!CreatePipe(&hOutputRead, &hOutputWrite, &saAttr, 0)) {
32 fprintf(stderr, "[PTY] Failed to create output pipe. Error: %lu\n", GetLastError());
33 CloseHandle(hInputRead);
34 CloseHandle(hInputWrite);
35 return false;
36 }
37
38 COORD terminalSize = {(short)columns, (short)rows};
39 HRESULT hr = CreatePseudoConsole(terminalSize, hInputRead, hOutputWrite, 0, &state->hPC);
40
41 if (FAILED(hr)) {
42 fprintf(stderr, "[PTY] Failed to create ConPTY. HRESULT: 0x%08lX\n", hr);
43 CloseHandle(hInputRead);
44 CloseHandle(hInputWrite);
45 CloseHandle(hOutputRead);
46 CloseHandle(hOutputWrite);
47 return false;
48 }
49
50 CloseHandle(hInputRead);
51 CloseHandle(hOutputWrite);
52
53 state->hInputWrite = hInputWrite;
54 state->hOutputRead = hOutputRead;
55
56 return true;
