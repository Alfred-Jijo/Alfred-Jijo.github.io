105/*
106 * Failure Capture Buffer
107 *
108 * EXPECT() fires inside the test function. Rather than printing immediately
109 * (which interleaves with PASS/FAIL lines), failures are buffered and then
110 * printed indented beneath the failing test name.
111 */
112
113#define TR_MAX_FAILURES_PER_TEST 16
114#define TR_FAILURE_MSG_LEN 256
115
116typedef struct {
117 char msg[TR_FAILURE_MSG_LEN];
118} _TrFailureMsg;
119
120static _TrFailureMsg _tr_fail_buf[TR_MAX_FAILURES_PER_TEST];
121static int _tr_fail_count = 0;
122
123static void
124_tr_record_failure(const char *file, int line, const char *expr)
125{
126 if (_tr_fail_count >= TR_MAX_FAILURES_PER_TEST) return;
127
128 /* Strip to filename only. */
129 const char *p = file;
130 const char *slash = strrchr(file, '/');
131 const char *bslash = strrchr(file, '\\');
132 if (slash && slash > p) p = slash + 1;
133 if (bslash && bslash > p) p = bslash + 1;
134
135 snprintf(_tr_fail_buf[_tr_fail_count].msg, TR_FAILURE_MSG_LEN,
136 "%s:%d %s", p, line, expr);
137 _tr_fail_count++;
138}

188static void
189_tr_run_test(_TrTestFn fn, const char *name)
190{
191 _tr_fail_count = 0;
192 int fails_before = _tr_failed;
193
194 clock_t t0 = clock();
195 fn();
196 clock_t t1 = clock();
197 long ms = (long)(((double)(t1 - t0) / CLOCKS_PER_SEC) * 1000.0);
198
199 _tr_total++;
200 _tr_suite_total++;
201
202 if (_tr_failed == fails_before) {
203 _tr_passed++;
204 printf(" %sPASS%s %-34s %s(%ldms)%s\n",
205 TR_GREEN, TR_RESET, name, TR_DIM, ms, TR_RESET);
206 } else {
207 _tr_suite_failed++;
208 printf(" %sFAIL%s %-34s %s(%ldms)%s\n",
209 TR_RED, TR_RESET, name, TR_DIM, ms, TR_RESET);
210 for (int i = 0; i < _tr_fail_count; i++) {
211 printf(" %s%s%s\n",
212 TR_YELLOW, _tr_fail_buf[i].msg, TR_RESET);
213 }
214 }
215}
