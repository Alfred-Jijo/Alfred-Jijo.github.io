23static void *
24arena_push_internal(Arena *arena, usize size, usize alignment)
25{
26 ASSERT(IS_POW2(alignment) && "alignment must be a non-zero power of 2");
27
28 // Align the current position up to the requested boundary.
29 usize current_pos = ALIGN_UP_POW2(arena->pos, alignment);
30 usize next_pos = current_pos + size;
31
32 // Guard: prevent overflowing past the total reserved capacity.
33 if (next_pos > arena->cap) {
34 ASSERT(0 && "Arena OOM: allocation exceeds reserved capacity");
35 return NULL;
36 }
37
38 // Commit more physical RAM if the allocation would exceed what we have.
39 if (next_pos > arena->cmt) {
40 // Round the required commit up to our chunk granularity.
41 usize needed = next_pos - arena->cmt;
42 usize cmt_size = ALIGN_UP_POW2(needed, ARENA_COMMIT_SIZE);
43 // Clamp so we never try to commit past the reservation ceiling.
44 usize cmt_clamped = MIN(cmt_size, arena->cap - arena->cmt);
45
46 b32 ok = os_mem_commit(arena->base + arena->cmt, cmt_clamped);
47 if (!ok) {
48 ASSERT(0 && "Arena OOM: OS refused to commit physical RAM");
49 return NULL;
50 }
51
52 arena->cmt += cmt_clamped;
53 }
54
55 void *result = arena->base + current_pos;
56 arena->pos = next_pos;
57 return result;
58}

110arena_pop_to(Arena *arena, usize pos)
111{
112 // A pos ahead of the current position is almost certainly a caller bug.
113 ASSERT(pos <= arena->pos && "arena_pop_to: pos is ahead of current position");
114 // Clamp defensively so release builds still behave safely.
115 usize clamped = MIN(pos, arena->pos);
116 arena->pos = clamped;
117}
