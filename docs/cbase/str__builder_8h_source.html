132function inline void
133sb_append_fmt(StringBuilder *sb, Arena *arena, const char *fmt, ...)
134{
135 va_list args;
136
137 // First pass: measure the required length (vsnprintf returns bytes
138 // excluding the NUL terminator).
139 va_start(args, fmt);
140 int len = vsnprintf(NULL, 0, fmt, args);
141 va_end(args);
142
143 if (len <= 0) return;
144
145 // Allocate exactly len + 1 bytes from the arena (for the NUL).
146 char *buf = (char *)arena_push_zero(arena, (usize)len + 1);
147 ASSERT_NOT_NULL(buf);
148
149 // Second pass: write the formatted string into the arena buffer.
150 va_start(args, fmt);
151 vsnprintf(buf, (usize)len + 1, fmt, args);
152 va_end(args);
153
154 // Append a slice of exactly len bytes (excluding NUL) to the list.
155 sb_append(sb, arena, str8((u8 *)buf, (usize)len));
156}

174function inline String8
175sb_join(StringBuilder *sb, Arena *out_arena)
176{
177 if (sb->total_size == 0) return (String8){0};
178
179 u8 *buf = (u8 *)arena_push(out_arena, sb->total_size);
180 ASSERT_NOT_NULL(buf);
181
182 usize offset = 0;
183 for (String8Node *n = sb->first; n != NULL; n = n->next) {
184 MEM_COPY(buf + offset, n->string.str, n->string.size);
185 offset += n->string.size;
186 }
187
188 return str8(buf, sb->total_size);
189}
