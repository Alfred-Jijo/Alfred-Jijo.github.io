136function inline String8
137sb_join(StringBuilder *sb, MemArena *out_arena)
138{
139 if (0 == sb->total_size) {
140 return (String8){0};
141 }
142
143 u8 *buf = (u8 *)mem_arena_push(out_arena, sb->total_size);
144 ASSERT_NOT_NULL(buf);
145
146 usize offset = 0;
147 for (String8Node *n = sb->first; n != NULL; n = n->next) {
148 MEM_COPY(buf + offset, n->string.str, n->string.size);
149 offset += n->string.size;
150 }
151
152 return str8(buf, sb->total_size);
153}
