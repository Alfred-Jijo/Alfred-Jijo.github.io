67// The zero_init flag is encoded directly inside the Allocator's ctx pointer as
68// a uintptr_t (0 = no zero-init, 1 = zero-init). This avoids a heap allocation
69// for the context object, eliminating any leak risk and the need for a destroy
70// function. The value is never dereferenced — only read as an integer.
71
72function inline void *
73_gp_alloc_fn(void *ctx, usize size)
74{
75 if ((uintptr_t)ctx) {
76 return calloc(1, size);
77 }
78 return malloc(size);
79}

81function inline void
82_gp_free_fn(void *ctx, void *ptr, usize size)
83{
84 (void)ctx;
85 (void)size;
86 free(ptr);
87}

89function inline void *
90_gp_realloc_fn(void *ctx, void *ptr, usize old_size, usize new_size)
91{
92 void *np = realloc(ptr, new_size);
93
94 // Zero only the *new* tail if caller wants zero-init
95 if (np && (uintptr_t)ctx && new_size > old_size) {
96 memset((char *)np + old_size, 0, new_size - old_size);
97 }
98 return np;
99}
