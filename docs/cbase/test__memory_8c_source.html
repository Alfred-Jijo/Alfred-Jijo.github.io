4TEST_CASE(arena_allocation)
5{
6 Arena a = arena_create(GB(1));
7 EXPECT_FATAL(a.base != NULL);
8 EXPECT(a.cap == GB(1));
9 EXPECT(a.pos == 0);
10 EXPECT(a.cmt == 0);
11
12 // Push a single int. The allocator aligns the *start* of this allocation
13 // to ARENA_DEFAULT_ALIGN (8), but pos only advances by sizeof(int) == 4
14 // bytes of actual data. So pos == 4, not 8.
15 int *num = PUSH_STRUCT(&a, int);
16 EXPECT_FATAL(num != NULL);
17 *num = 42;
18 EXPECT(a.pos == 4);
19 EXPECT(*num == 42);
20
21 // Physical RAM should now be committed in at least one 64 KB chunk.
22 EXPECT(a.cmt >= ARENA_COMMIT_SIZE);
23
24 // Push an array and verify zero-initialization.
25 int *arr = PUSH_ARRAY(&a, int, 10);
26 EXPECT_FATAL(arr != NULL);
27 EXPECT(arr[0] == 0);
28 EXPECT(arr[9] == 0);
29
30 // pos was 4 after the int. Next alloc aligns up to 8, then adds 10*4=40 -> pos == 48.
31 EXPECT(a.pos == 48);
32
33 arena_clear(&a);
34 EXPECT(a.pos == 0); // pos resets; committed RAM stays mapped.
35
36 arena_release(&a);
37 EXPECT(a.base == NULL);
38 EXPECT(a.cap == 0);
39}
