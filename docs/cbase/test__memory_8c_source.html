6TEST_CASE(mem_arena_allocation)
7{
8 MemArena a = mem_arena_create(GB(1));
9 EXPECT_FATAL(a.base != NULL);
10 EXPECT(a.reserve == GB(1));
11 EXPECT(a.pos == 0);
12 EXPECT(a.commit == 0);
13
14 // Push a single int. The allocator aligns the *start* of this allocation
15 // to MEM_ARENA_ALIGNMENT (8), but pos only advances by sizeof(int) == 4
16 // bytes of actual data. So pos == 4, not 8.
17 int *num = PUSH_STRUCT(&a, int);
18 EXPECT_FATAL(num != NULL);
19 *num = 42;
20 EXPECT(a.pos == 4);
21 EXPECT(*num == 42);
22
23 // Physical RAM should now be committed in at least one 64 KB chunk.
24 EXPECT(a.commit >= KB(64));
25
26 // Push an array and verify zero-initialization.
27 int *arr = PUSH_ARRAY(&a, int, 10);
28 EXPECT_FATAL(arr != NULL);
29 EXPECT(arr[0] == 0);
30 EXPECT(arr[9] == 0);
31
32 // pos was 4 after the int. Next alloc aligns up to 8, then adds 10*4=40 -> pos == 48.
33 EXPECT(a.pos == 48);
34
35 mem_arena_clear(&a);
36 EXPECT(a.pos == 0); // pos resets; committed RAM stays mapped.
37
38 mem_arena_release(&a);
39 EXPECT(a.base == NULL);
40 EXPECT(a.reserve == 0);
41}
