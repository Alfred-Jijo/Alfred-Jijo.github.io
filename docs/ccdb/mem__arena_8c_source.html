35void *
36arena_push(Arena *arena, usize size)
37{
38 // Calculate the new position, aligned to an 8-byte boundary for CPU performance
39 usize current_pos = ALIGN_UP_POW2(arena->pos, 8);
40 usize next_pos = current_pos + size;
41
42 // Prevent buffer overflows past the absolute capacity
43 if (next_pos > arena->cap) {
44 ASSERT(0 && "Arena out of memory (exceeded capacity)!");
45 return NULL;
46 }
47
48 // Do we need to ask the OS for more physical RAM?
49 if (next_pos > arena->cmt) {
50 // Calculate how much to commit (rounded up to our chunk size)
51 usize cmt_size = ALIGN_UP_POW2(next_pos - arena->cmt, ARENA_COMMIT_SIZE);
52
53 // Ensure we don't accidentally try to commit past the total capacity
54 usize cmt_clamped = MIN(cmt_size, arena->cap - arena->cmt);
55
56 b32 success = os_mem_commit(arena->base + arena->cmt, cmt_clamped);
57 if (!success) {
58 ASSERT(0 && "Arena out of memory (OS failed to commit physical RAM)!");
59 return NULL;
60 }
61
62 arena->cmt += cmt_clamped;
63 }
64
65 // Perform the allocation
66 void *result = arena->base + current_pos;
67 arena->pos = next_pos;
68
69 return result;
70}
