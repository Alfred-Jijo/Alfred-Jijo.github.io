/**
 * sawmill.h
 *
 * A lightweight, stb style, thread-safe C99 logging library.
 *
 * Features:
 * - Leveled logging (TRACE through FATAL)
 * - Thread-safety via user-defined locking callbacks
 * - Multiple destinations: Console (colored), Files, and Custom Callbacks
 * - Automatic metadata capture (File, Line, Timestamp)
 */

#ifndef SAWMILL_H
#define SAWMILL_H

#include <stdio.h>
#include <stdarg.h>
#include <stdbool.h>
#include <time.h>
#include <stdlib.h>
#include <string.h>

#define SAWMILL_H_VERSION "1.1.4"

// LOG LEVELS
enum {
        SM_TRACE = 0,
        SM__DEBUG,
        SM_INFO,
        SM_WARN,
        SM_ERROR,
        SM_FATAL
};

// Event Struvct passed to Callbacks
typedef struct {
        va_list     ap   ;      // Variadic arguments
        const char *fmt  ;      // Format string
        const char *file ;      // Filename
        struct tm  *time ;      // Parsed time
        void       *udata;      // User data provided in add_callback
        int         line ;      // Line number
        int         level;      // Log level
} sm_Event;

// Callbacks
typedef void (*sm_LogFn)(sm_Event *ev);
typedef void (*sm_LockFn)(bool lock, void *udata);

// Set the lock function (pass NULL to disable locking)
void sm_set_lock(sm_LockFn fn, void *udata);

// Set the global logging level (logs below this level are ignored)
void sm_set_level(int level);

// Enable or disable quiet mode (true = no output to stdout/stderr)
void sm_set_quiet(bool enable);

// Add a file pointer to log to (e.g., an open log file)
void sm_add_fp(FILE *fp, int level);

// Add a custom callback function (useful for writing to databases or custom formats)
void sm_add_callback(sm_LogFn fn, void *udata, int level);

// Core Logging Function
void sm_log(int level, const char *file, int line, const char *fmt, ...);

// Macros 
#define sm_trace(...) sm_log(SM_TRACE, __FILE__, __LINE__, __VA_ARGS__)
#define sm_debug(...) sm_log(SM__DEBUG, __FILE__, __LINE__, __VA_ARGS__)
#define sm_info(...)  sm_log(SM_INFO,  __FILE__, __LINE__, __VA_ARGS__)
#define sm_warn(...)  sm_log(SM_WARN,  __FILE__, __LINE__, __VA_ARGS__)
#define sm_error(...) sm_log(SM_ERROR, __FILE__, __LINE__, __VA_ARGS__)
#define sm_fatal(...) sm_log(SM_FATAL, __FILE__, __LINE__, __VA_ARGS__)

#define sm_trace_if(cond, ...) if (cond) sm_log(SM_TRACE, __FILE__, __LINE__, __VA_ARGS__)
#define sm_debug_if(cond, ...) if (cond) sm_log(SM__DEBUG, __FILE__, __LINE__, __VA_ARGS__)
#define sm_info_if(cond, ...)  if (cond) sm_log(SM_INFO,  __FILE__, __LINE__, __VA_ARGS__)
#define sm_warn_if(cond, ...)  if (cond) sm_log(SM_WARN,  __FILE__, __LINE__, __VA_ARGS__)
#define sm_error_if(cond, ...) if (cond) sm_log(SM_ERROR, __FILE__, __LINE__, __VA_ARGS__)
#define sm_fatal_if(cond, ...) if (cond) sm_log(SM_FATAL, __FILE__, __LINE__, __VA_ARGS__)

#define sm_assert(cond, ...)                                                                   \
        do {                                                                                   \
                if (!(cond)) {                                                                   \
                        sm_log(SM_FATAL, __FILE__, __LINE__, "ASSERTION FAILED: " #cond); \
                        sm_log(SM_FATAL, __FILE__, __LINE__, __VA_ARGS__);                   \
                        abort();                                                           \
                }                                                                           \
        } while (0)

#ifdef SAWMILL_IMPLEMENTATION

// Configuration Constants
#define MAX_CALLBACKS 32

typedef struct {
        sm_LogFn fn;
        void *udata;
        int level;
} Callback;

static struct {
        void *udata;
        sm_LockFn lock;
        int level;
        bool quiet;
        Callback callbacks[MAX_CALLBACKS];
} L; // Internal library state

// Color codes for console output
static const char *level_strings[] = {
        "TRACE", "DEBUG", "INFO", "WARN", "ERROR", "FATAL"
};

static const char *level_colors[] = {
        "\x1b[94m", // Trace - Light Blue
        "\x1b[36m", // Debug - Cyan
        "\x1b[32m", // Info  - Green
        "\x1b[33m", // Warn  - Yellow
        "\x1b[31m", // Error - Red
        "\x1b[35m"  // Fatal - Magenta
};

static void lock(void) {
        if (L.lock) { L.lock(true, L.udata); }
}

static void unlock(void) {
        if (L.lock) { L.lock(false, L.udata); }
}

static void stdout_callback(sm_Event *ev) {
        char buf[16];
        buf[strftime(buf, sizeof(buf), "%H:%M:%S", ev->time)] = '\0';

        // Format: Time [LEVEL] File:Line: Message
        fprintf(
                        ev->udata,
                        "%s %s%-5s\x1b[0m \x1b[90m%s:%d:\x1b[0m ",
                        buf,
                        level_colors[ev->level],
                        level_strings[ev->level],
                        ev->file,
                        ev->line
               );

        vfprintf(ev->udata, ev->fmt, ev->ap);
        fprintf(ev->udata, "\n");
        fflush(ev->udata);
}

static void file_callback(sm_Event *ev) {
        char buf[64];
        buf[strftime(buf, sizeof(buf), "%Y-%m-%d %H:%M:%S", ev->time)] = '\0';

        // Format but with colour: Time [LEVEL] File:Line: Message
        fprintf(
                        ev->udata,
                        "%s [%-5s] %s:%d: ",
                        buf,
                        level_strings[ev->level],
                        ev->file,
                        ev->line
               );

        vfprintf(ev->udata, ev->fmt, ev->ap);
        fprintf(ev->udata, "\n");
        fflush(ev->udata);
}

/**
 * Actual API what not
 */
void sm_set_lock(sm_LockFn fn, void *udata) {
        L.lock = fn;
        L.udata = udata;
}

void sm_set_level(int level) {
        L.level = level;
}

void sm_set_quiet(bool enable) {
        L.quiet = enable;
}

void sm_add_callback(sm_LogFn fn, void *udata, int level) {
        for (int i = 0; i < MAX_CALLBACKS; i++) {
                if (!L.callbacks[i].fn) {
                        L.callbacks[i] = (Callback) { fn, udata, level };
                        return;
                }
        }
}

void sm_add_fp(FILE *fp, int level) {
        sm_add_callback(file_callback, fp, level);
}

void sm_log(int level, const char *file, int line, const char *fmt, ...) {
        // Initial check for global level logic should be unlocked
        // This prevents locking overhead for suppressed logs unless individual callbacks need lower levels 

        sm_Event ev = {
                .fmt   = fmt,
                .file  = file,
                .line  = line,
                .level = level,
        };

        lock();

        // Get current time
        time_t t = time(0);
        ev.time = localtime(&t);

        // Log to stdout/stderr unless quiet
        if (!L.quiet && level >= L.level) {
                ev.udata = (level >= SM_ERROR) ? stderr : stdout;
                va_start(ev.ap, fmt);
                stdout_callback(&ev);
                va_end(ev.ap);
        }

        // Process callbacks and file pointers
        for (int i = 0; i < MAX_CALLBACKS && L.callbacks[i].fn; i++) {
                Callback *cb = &L.callbacks[i];
                if (level >= cb->level) {
                        ev.udata = cb->udata;
                        va_start(ev.ap, fmt);
                        cb->fn(&ev);
                        va_end(ev.ap);
                }
        }

        unlock();
}

#endif // SAWMILL_IMPLEMENTATION
#endif // SAWMILL_H
